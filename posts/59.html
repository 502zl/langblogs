排序算法小结

图解总结

先上一个图

术语说明：

  • 稳定 ：如果a原本在b前面，而a=b，排序之后a仍然在b的前面；
  • 不稳定 ：如果a原本在b的前面，而a=b，排序之后a可能会出现在b的后面；
  • 内排序 ：所有排序操作都在内存中完成；
  • 外排序 ：由于数据太大，因此把数据放在磁盘中，而排序通过磁盘和内存的数据传输才能进行；
  • 时间复杂度 ： 一个算法执行所耗费的时间规模。
  • 空间复杂度 ：运行完一个程序所需内存规模。

比较排序：
在排序的最终结果里，元素之间的次序依赖于它们之间的比较。每个数都必须和其他数进行比较，才能确定自己的位置。
快速排序、归并排序、堆排序、冒泡排序等属于比较排序。

非比较排序：
非比较排序是通过确定每个元素之前，应该有多少个元素来排序。针对数组arr,计算arr[i]之前有多少个元素，则唯一确定了arr[i]在排序后数组中的位置。
计数排序、基数排序、桶排序则属于非比较排序

具体实现及优化

选择排序

思路：将每个元素标记为小（大）的元素，并向右遍历，在一趟比较完，把真正的最小（大）的元素和标记互换，并把标记向右移动，直到遍历到倒数第二个元素。

图解：

基础版：

function selectSort(arr) {
  if(arr == null || arr.length < 2 ) {
    return;
  }

  for(let i = 0; i < arr.length - 1; i++) {
    // 最值标记
    let minIndex = i;
    for(let j = i + 1; j < arr.length; j++) {
      count++;
      if(arr[j] < arr[minIndex]) {
        minIndex = j
      }
    }
    // 判断标记是否改变
    if(minIndex !== i) {
      let temp = arr[i]
      arr[i] = arr[minIndex]
      arr[minIndex] = temp;
    }
  }
}

let arr1 = [6, 3, 8, 9, 1, 5]
let count = 0;
selectSort(arr1) 
console.log(arr1) // [1, 3, 5, 6, 8, 9]
console.log("count = ", count) // count = 15

优化：
选择排序的优化思路一般是在一趟遍历中，同时找出最大值与最小值，放到数组两端，这样就能将遍历的趟数减少一半

上代码:

function swap(arr, front, later) {
  let temp = arr[front]
  arr[front] = arr[later]
  arr[later] = temp
}

function selectSort(arr) {
  if(arr == null || arr.length < 2 ) {
    return;
  }

  let leftFlag = 0
  let rightFlag = arr.length - 1

  while(leftFlag < rightFlag) {
    let minIndex = leftFlag;
    let maxIndex = rightFlag;
    for(let i = leftFlag; i <= rightFlag; i++) {
      count++
      if(arr[i] < arr[minIndex]) {
        minIndex = i
      }
      if(arr[i] > arr[maxIndex]) {
        maxIndex = i
      }
    }

    if(minIndex === maxIndex) break;

    if(minIndex !== leftFlag) {
      swap(arr, leftFlag, minIndex)
    }

    // 上面循环中，如果找到这个位置为最大的话，就不需要重复交换了
    // 因为此时，leftFlag被更新，已经为最小，而最大值，被换到了minIndex主
    console.log(maxIndex)
    console.log(minIndex)

    if(maxIndex === leftFlag) {
      maxIndex = minIndex
    }

    if(maxIndex !== rightFlag ) {
      swap(arr, maxIndex, rightFlag)
    }
    leftFlag++
    rightFlag--
  }
}

let arr1 = [6, 3, 8, 9, 1, 5]
let count = 0;
selectSort(arr1) 
console.log(arr1) // [1, 3, 5, 6, 8, 9]
console.log("count = ", count) // count = 12

冒泡排序

思路：相邻的元素两两比较，较大的数下沉，较小的数冒起来

图解：

基础版：

function swap(arr, front, later) {
  let temp = arr[front]
  arr[front] = arr[later]
  arr[later] = temp
}

function bubbleSort(arr) {
  if(arr == null || arr.length < 2) {
    return;
  }

  for(let i = 0; i < arr.length - 1; i++) {
    for(let j = 0; j < arr.length - i - 1; j++){
      count++
      if(arr[j] > arr[j+1]) {
        swap(arr, j, j+1)
      }
    }
  }
}

let arr1 = [6, 3, 8, 9, 1, 5]
var count = 0;
bubbleSort(arr1) 
console.log(arr1) // [1, 3, 5, 6, 8, 9]
console.log("count = ", count) // count = 15

在基础版中，当一个数组已经是有序的情况，还是会遍历那么多次
所以，在优化版中，需要解决这个问题

优化版：

function swap(arr, front, later) {
  let temp = arr[front]
  arr[front] = arr[later]
  arr[later] = temp
}

function bubbleSort(arr) {
  if(arr == null || arr.length < 2) {
    return;
  }

  for(let i = 0; i < arr.length - 1; i++) {
    let isSwap = false;
    for(let j = 0; j < arr.length - i - 1; j++){
      count++;
      if(arr[j] > arr[j+1]) {
        swap(arr, j, j+1);
        isSwap = true;
      }
    }
    if(!isSwap) {
      break;
    }
  }
}

let arr1 = [1, 2, 4, 3, 5, 6]
var count = 0;
bubbleSort(arr1) 
console.log(arr1) // [1, 2, 3, 4, 5, 6]
console.log("count = ", count) // count = 9

这个在循环内部加标志位的方法只能在第二趟中没有交换了，才能判断出当前已经是有序了，所以还是会多走一趟。

所以，想办法继续优化。。。

思路：因为每次都能冒出一个最大的值放在数组的尾部，所以在第一趟时，可以确定最大的数，且会把它数组的末尾，所以进行下一次比较时，比较到倒数第二个位置就行；在第二趟时，可以确定第二大的数，会把它数组的末尾倒数第二位，进行下一次比较时，比较到倒数第三个位置就行，以此类似。所以我们可以记录最后交换的位置，下一次比较进行到这个位置之前一位就行了，因为，最后交换的位置之后都是有序的了，这样就可以在次数上做优化。

精良版：

function swap(arr, front, later) {
  let temp = arr[front]
  arr[front] = arr[later]
  arr[later] = temp
}

function bubbleSort(arr) {
  if(arr == null || arr.length < 2) {
    return;
  }

  let lastSwapIndex = arr.length - 1;
  for(let i = 0; i < arr.length - 1; i++) {
    let isSwap = false;
    for(let j = 0; j < lastSwapIndex; j++){
      count++;
      if(arr[j] > arr[j+1]) {
        swap(arr, j, j+1);
        isSwap = true;
        lastSwapIndex = j
      }
    }
    if(!isSwap) {
      break;
    }
  }
}

let arr1 = [1, 2, 4, 3, 5, 6]
var count = 0;
bubbleSort(arr1) 
console.log(arr1) // [1, 2, 3, 4, 5, 6]
console.log("count = ", count) // count = 5

冒泡的其他版本：双向冒泡遍历（鸡尾酒排序）

就是从两头开始找，把最值放在两端，上面的都是从一头开始找，但我感觉这玩意和改进版的选择排序很相似，只是是两个遍历，从两端开始找最值，然后替换到开头结尾。

插入排序

思路：构建有序序列，对于未排序的数组元素，在已排好序的序列中从后往前扫描，并插入到合适的位置，使这段依然是有序的

图解：

function insertSort(arr) {
  if (arr == null || arr.length < 2) return;
  for (let i = 1; i < arr.length; i++) {
    let currentIndex = i;
    let currentValue = arr[i];

    // 找到该插入的位置，注意边界条件
    while (currentIndex > 0 && currentValue < arr[currentIndex - 1]) {
      count++;
      arr[currentIndex] = arr[currentIndex - 1];
      currentIndex--;
    }
    arr[currentIndex] = currentValue;
  }
}

let arr1 = [6, 3, 8, 9, 1, 5];
var count = 0;
insertSort(arr1);
console.log(arr1); // [ 1, 3, 5, 6, 8, 9 ]
console.log("count = ", count); // count = 8

优化：在查找那一块，用的是暴力查找，且因为要查找的序列为有序的，所以我们可以使用一些更为高效的查找算法，比如二分查找之类的

快速排序

快排是冒泡的一种改进
思路：每趟排序时选出一个基准值，然后将所有元素与该基准值比较，并按大小分成左右两堆，然后递归执行该过程，直到所有元素都完成排序。

图解：

快排比较难，写个步骤…
快速排序使用分治法来把一个串（list）分为两个子串（sub-lists）:

  1. 从数列中挑出一个元素，称为 “基准”（pivot ）；
  2. 重新排序数列，所有元素比基准值小的摆放在基准前面，所有元素比基准值大的摆在基准的后面（相同的数可以到任一边）。在这个分区退出之后，该基准就处于数列的中间位置。这个称为分区（partition）操作；
  3. 递归地（recursive）把小于基准值元素的子数列和大于基准值元素的子数列排序。
function swap(arr, front, later) {
  let temp = arr[front];
  arr[front] = arr[later];
  arr[later] = temp;
}

function quickSort(arr) {
  if (arr == null || arr.length < 2) {
    return;
  }
  sortHandle(arr, 0, arr.length - 1);
}

function sortHandle(arr, start, end) {
  if (start >= end) return;

  let pivotIndex = partition(arr, start, end);
  if (pivotIndex > start) sortHandle(arr, start, pivotIndex - 1);
  if (pivotIndex < end) sortHandle(arr, pivotIndex + 1, end);
}

// 这种方法把每一个分区的第一个元素作为基底
function partition(arr, start, end) {
  let swapIndex = start;
  let pivot = arr[start];
  start++;
  while(start <= end) {
    if(arr[start] < pivot) {
      swap(arr, swapIndex, start);
      swapIndex = start;
    }
    start++;
  }
  return swapIndex;
}

const arr1 = [6, 3, 8, 9, 1, 5];
var count = 0;
quickSort(arr1);
console.log(arr1); // [ 1, 3, 5, 6, 8, 9 ]
console.log("count = " + count); // count = 9

分析：
当选择头/尾元素作为基值的方案下，当数组已经是有序的情况下，此时的快排就是没有任何优化的冒泡了，效率非常糟糕
在多次数据测试下，发现能使快排效率性能较好的两种基值选取方案：1、随机选取；2、三数取中法

演示一下随机选取基值的方案：

function swap(arr, front, later) {
  let temp = arr[front];
  arr[front] = arr[later];
  arr[later] = temp;
}

function quickSort(arr) {
  if (arr == null || arr.length < 2) {
    return;
  }
  sortHandle(arr, 0, arr.length - 1);
}

function sortHandle(arr, start, end) {
  if (start >= end) return;

  let pivotIndex = partition(arr, start, end);
  if (pivotIndex > start) sortHandle(arr, start, pivotIndex - 1);
  if (pivotIndex < end) sortHandle(arr, pivotIndex + 1, end);
}

function partition(arr, start, end) {
  let pivotIndex = start + Math.ceil(Math.random() * (end - start));
  swap(arr, start, pivotIndex)
  let swapIndex = start;
  let pivot = arr[start];
  for (let i = start + 1; i <= end; i++) {
    count++;
    if (arr[i] < pivot) {
      swapIndex++;
      swap(arr, swapIndex, i);
    }
  }
  swap(arr, start, swapIndex);
  return swapIndex;
}

const arr1 = [6, 3, 8, 9, 1, 5];
var count = 0;
quickSort(arr1);
console.log(arr1); // [ 1, 3, 5, 6, 8, 9 ]
console.log("count = " + count); // count = 9

堆排序

基本思想：

1.首先将待排序的数组构造成一个大根堆，此时，整个数组的最大值就是堆结构的顶端
2.将顶端的数与末尾的数交换，此时，末尾的数为最大值，剩余待排序数组个数为n-1
3.将剩余的n-1个数再构造成大根堆，再将顶端数与n-1位置的数交换，如此反复执行，便能得到有序数组

代码：

function swap(nums, index1, index2) {
  let temp = nums[index1];
  nums[index1] = nums[index2];
  nums[index2] = temp;
}

function buildHeap(nums) {
  let len = nums.length;
  if(len < 2) return;
  for(let i = Math.floor(len/2); i >= 0; i--) {
    heapify(nums, len, i);
  }
}

function heapify(nums, len, curIndex) {
  let maxChildIndex = 2 * curIndex + 1;
  let baseVal = nums[curIndex];

  while(maxChildIndex < len) {
    if(maxChildIndex+1 < len && nums[maxChildIndex] < nums[maxChildIndex+1]) {
      maxChildIndex += 1;
    }
    if(nums[maxChildIndex] > nums[curIndex]) {
     swap(nums, curIndex, maxChildIndex);
      curIndex = maxChildIndex;
      maxChildIndex = 2 * curIndex + 1;
    } else {
      break;
    }
  }
}

function heapSort(nums) {
  let len = nums.length;
  if(len < 2) return;

  buildHeap(nums);
  
  for(let i = len-1; i >= 0; i--) {
    swap(nums, 0, i);
    heapify(nums, i, 0);
  }
}

let arr = [9, 8, 1, 10, 3];
heapSort(arr);
console.log(arr); // [ 1, 3, 8, 9, 10 ]
