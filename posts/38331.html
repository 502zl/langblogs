JavaScript 改变函数执行上下文（改变函数中this的指向）

背景

geigei, 为什么要改变函数执行上下文？

嘿嘿，你一定听过一个词语 –> 移花接木（记得小时候看过一个电视剧里面有一个招数就叫这个，哈哈哈，扯远了）

想了一个场景一：

const cat = {
  name: "mimi",
  say: function () {
    console.log(this.name, "：喵喵喵");
  }
};

const dog = {
  name: "旺财",
};

// 让mimi说话
cat.say();  // log："mimi：喵喵喵"

// 让一个啥也不是的变量来模仿mimi的说话(注意say() 和 say() 的区别)
let temp = cat.say;
temp(); // log: "undefined: 喵喵喵"

// 那么这时，我们想让旺财「临时」来模仿mimi的说话方式，但是又不想去修改旺财自己的本性（不去修改对象本身）
// （就是让dog去使用cat.say()的函数执行上下文）
// 分析代码，就是要改变say()里面的this，但如何实现呢？？？

// call(), apply(), bind() 请求出战！！！
cat.say.call(dog);   // log: "旺财 ：喵喵喵"
cat.say.apply(dog);  // log: "旺财 ：喵喵喵"
cat.say.bind(dog)(); // log: "旺财 ：喵喵喵"

// 看看旺财，是否被污染，并没有
console.log(dog)  // log "{ name: '旺财' }"

// "绝绝子"

看了这个场景应该有思路了吧。。。

再看看场景二：
现实中的需求：在满足某种情况下，需要执行某些特殊的回调操作


const person1 = {
  name: "xiaoli",
  age: 20,
};

// 一个公用的特殊操作
function handle() {
  console.log(this.name, "，恭喜你，你已经是大人了");
}

function isAdult(callback, person) {
  // 回调处理(看两种方式的区别)
  if (person.age >= 18) {
    callback();
    callback.call(person1);
  }
}

// 去判断一个人是否成年
// 注意，如果成年，去执行回调handle操作，但这里只是简单的log，一般是比较复杂的操作
isAdult(handle, person1)

// 注意看输出结果
/*
log： 
    undefined ，恭喜你，你已经是大人了
    xiaoli ，恭喜你，你已经是大人了
*/

细说一下call()，apply()，bind()

函数名称参数格式是否立即调用
call()(targetObj, arg1, arg2, … )是
apply()(targetObj, [arg1, arg2, …])是
bind()(targetObj, arg1, arg2, …否，返回一个已经改变了this的函数

call()

其第一个参数接收函数体内的this要重新指向的对象，第二个及之后的参数为需要传进函数内的参数

关于参数：

// 关于call第一个参数
function handle() {
    console.log(this);
}

// 不传参，或基本类型时
handle.call() // node环境下，this指向global，浏览器下指向window
handle.call(null) // node环境下，this指向global，浏览器下指向window
handle.call(undefined) // node环境下，this指向global，浏览器下指向window
handle.call(1) // this指向 [Number: 1]
handle.call('string') // this指向[String: 'string']
handle.call(true) // this指向[Boolean: true]


// 关于call第二个及之后的参数
var obj = {};
function changeArr(name, age) {
    this.name = name;
    this.age = age
}
changeArr.call(obj, "xiaomi", 20)
console.log(obj) // { name: 'xiaomi', age: 20 }

apply()

apply() 和 call() 效果完全相同，唯一不同的是传第二参上：apply传递的第二参数为数组，需要传进函数体内的参数全包含在里面

var obj = {};
function setSum() {
  this.sum = 0;
  
  for (let index in arguments) {
    this.sum += arguments[index];
  }
}
setSum.apply(obj, [1,2,3,4]);
console.log(obj) // log: { sum: 10 }

bind()

bind() 与 call() , apply() 的调用执行方式不同，call() 和 apply()方法会对目标函数进行调用执行，所以call() 和 apply() 无法用于事件绑定函数中，所以bind()的出现能解决这个问题。bind()不会调用执行目标函数，使用bind()之后，返回一个已经改变this指向的新绑定函数。

特点：

  1. 改变原函数的 this 指向，即绑定上下文，返回原函数的拷贝

  2. 当 绑定函数 被调用时，bind的额外参数将置于实参之前传递给被绑定的方法。

  3. 注意，一个 绑定函数 也能使用 new 操作符创建对象,这种行为就像把原函数当成构造器，thisArg 参数无效。也就是 new 操作符修改 this 指向的优先级更高。

var obj = {};
function changeArr(name, age) {
  this.name = name;
  this.age = age;
}
// 注意返回一个函数
let myFun = changeArr.bind(obj, "xiaomi", 20);
myFun();
console.log(obj); // log：{ name: 'xiaomi', age: 20 }

// 注意，这里用bind创建了实例
function setName(name) {
  this.name = name;
}

let obj1 = {};

let callback1 = setName.bind(obj1);
callback1("zl");
console.log(obj1); 
// {name： 'zl'}

let callback2 = setName.bind(obj1);
let obj2 = new callback2("zf");
console.log(obj2);
// {name: 'zf'}

手写实现

手写call()

// 需要理解原型和原型链 
Function.prototype.myCall = function(targetObj = global, ...args) {
  if(typeof this !== 'function') return;
  targetObj = targetObj == null ? global : targetObj;  // 此操作可以过滤掉 null 和 undefined
  const type = typeof targetObj;
  switch(type) {
    case 'number':
      targetObj = new Number(targetObj)
      break;
    case 'string':
      targetObj = new String(targetObj)
      break;
    case 'boolean':
      targetObj = new Boolean(targetObj)
  }
  targetObj.fun = this;
  let result = targetObj.fun(...args);
  let flag = delete targetObj.fun;
  return result;
}

var obj = {
  name: 'zl'
}
function testMyCall() {
  console.log(this);
  console.log('参数', arguments)
}

testMyCall.myCall(obj)

手写apply()

Function.prototype.myApply = function(targetObj = global, args) {
  if(typeof this !== 'function') return;
  targetObj = targetObj == null ? global : targetObj;
  const type = typeof targetObj;
  switch(type) {
    case 'number':
      targetObj = new Number(targetObj)
      break;
    case 'string':
      targetObj = new String(targetObj)
      break;
    case 'boolean':
      targetObj = new Boolean(targetObj)
  }
  targetObj.fun = this;
  let result = targetObj.fun(args)
  delete targetObj.fun;
  return result;
}
var obj = {
  name: 'zl'
}
function testMyApply() {
  console.log(this);
  console.log('参数', arguments)
}
testMyApply.myApply(obj, [1, 2, 3, 4])

手写 bind()

涉及到的知识点：使用new创建实例new做的事情，原型链，原型链继承，闭包

// 注意：
// 当使用new关键字调用testMyBind.myBind(this) 来创建对象的时候，此时fun.myBind(this) 的this应该指向fun()而非我们传入的targetObj。
// 所以当我们的myBind函数返回的函数被作为构造函数时，我们要让我们提供的targetObj失效。
Function.prototype.myBind = function (targetObj = global, ...args) {
  if (typeof this !== "function") return;
  let _this = this;
  let tempFn = function () {};
  // 维护原型关系
  if(this.prototype) {
    tempFn.prototype = this.prototype;
  }
  let backFn = function () {
    // 判断是否为创建实例
    targetObj = new.target ? this : targetObj;
    return _this.call(targetObj, ...args, ...arguments);
  };
  // 这里相当于原型链继承了
  backFn.prototype = new tempFn();
  backFn.prototype.constructor = backFn;
  return backFn;
};

var obj = {};
function testMyBind(name) {
  this.name = name;
  console.log(arguments);
}

var callBack = testMyBind.myBind(obj);
callBack("关羽");
console.log(obj.name);

// 注意这里
var obj1 = new callBack("刘备");  // 相当于 var ob1 = new testMyBind("刘备")
console.log(obj1);
